document.getElementById('productForm').addEventListener('submit', async (e) => { e.preventDefault(); const btn = document.getElementById('submitBtn'); btn.innerText = 'Enregistrement en cours...'; try { const name = document.getElementById('name').value; const price = document.getElementById('price').value; const file = document.getElementById('mediaFile').files[0]; console.log("Début de l'enregistrement..."); let mediaUrl = ''; if (file) { console.log("Upload du fichier en cours..."); const fileName = `${Date.now()}_${file.name}`; const { data, error } = await supabase.storage .from('media') // Vérifiez bien que le nom du bucket est 'media' .upload(fileName, file); if (error) throw new Error("Erreur Upload: " + error.message); const { data: publicUrlData } = supabase.storage.from('media').getPublicUrl(fileName); mediaUrl = publicUrlData.publicUrl; } console.log("Insertion dans la base de données..."); const { error: dbError } = await supabase .from('products') .insert([{ name, price, attributes: { mediaUrl } }]); if (dbError) throw new Error("Erreur Base: " + dbError.message); alert('Succès !'); location.reload(); } catch (err) { console.error(err); alert(err.message); // Ceci affichera l'erreur réelle sur votre téléphone btn.innerText = 'Enregistrer le produit'; } });
